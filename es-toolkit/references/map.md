# Map Operations Reference

Collection functions for iterating, filtering, and transforming arrays and objects.

## countBy

Groups collection elements by a key and counts elements in each group.

```typescript
function countBy<T>(arr: T[], iteratee: (item: T) => string): Record<string, number>
```

Counts the number of elements in an array grouped by the result of iterating function.

```typescript
import { countBy } from 'es-toolkit';

countBy([6.1, 4.2, 6.3], Math.floor);
// Returns: { '4': 1, '6': 2 }

countBy(['apple', 'banana', 'apricot', 'banana', 'apple'], (fruit) => fruit[0]);
// Returns: { 'a': 2, 'b': 2, 'a': 1 }
```

## every

Checks if all elements in a collection satisfy a condition.

```typescript
function every<T>(arr: T[], predicate: (item: T) => boolean): boolean
```

Returns `true` if all elements pass the predicate check.

```typescript
import { every } from 'es-toolkit';

every([1, 2, 3], (x) => x > 0);
// Returns: true

every([1, 2, 3], (x) => x > 2);
// Returns: false

every([], (x) => x > 0);
// Returns: true
```

## filter

Creates a new array with all elements that pass the test.

```typescript
function filter<T>(arr: T[], predicate: (item: T) => boolean): T[]
```

Returns a new array containing only elements that satisfy the predicate function.

```typescript
import { filter } from 'es-toolkit';

const users = [
  { name: 'favicon', active: true },
  { name: 'profile', active: false },
  { name: 'settings', active: true }
];

filter(users, (user) => user.active);
// Returns: [{ name: 'favicon', active: true }, { name: 'settings', active: true }]

filter([1, 2, 3, 4, 5], (x) => x % 2 === 0);
// Returns: [2, 4]
```

## findKey

Returns the key of the first element that satisfies the condition.

```typescript
function findKey<T>(obj: T, predicate: (value: T[keyof T]) => boolean): keyof T | undefined
```

```typescript
import { findKey } from 'es-toolkit';

const users = {
  barney: { age: 36, active: true },
  fred: { age: 40, active: false },
  pebbles: { age: 1, active: true }
};

findKey(users, (o) => o.age < 40);
// Returns: 'barney'

findKey(users, (o) => o.active === false);
// Returns: 'fred'
```

## findValue

Returns the value of the first element that satisfies the condition.

```typescript
function findValue<T>(arr: T[], predicate: (item: T) => boolean): T | undefined
```

```typescript
import { findValue } from 'es-toolkit';

const users = [
  { name: 'favicon', active: true },
  { name: 'profile', active: false },
  { name: 'settings', active: true }
];

findValue(users, (user) => !user.active);
// Returns: { name: 'profile', active: false }

findValue([1, 2, 3, 4], (x) => x > 3);
// Returns: 4
```

## forEach

Iterates over elements of a collection, executing a function for each element.

```typescript
function forEach<T>(arr: T[], callback: (item: T, index: number) => void): T[]
```

```typescript
import { forEach } from 'es-toolkit';

forEach([1, 2, 3], (value, index) => {
  console.log(`Index ${index}: ${value}`);
});
// Logs: 'Index 0: 1', 'Index 1: 2', 'Index 2: 3'

const result: number[] = [];
forEach([1, 2, 3], (value) => result.push(value * 2));
// result is: [2, 4, 6]
```

## hasValue

Checks if a value exists in an object.

```typescript
function hasValue<T>(obj: T, value: unknown): boolean
```

```typescript
import { hasValue } from 'es-toolkit';

hasValue({ a: 1, b: 2, c: 3 }, 2);
// Returns: true

hasValue({ a: 1, b: 2, c: 3 }, 5);
// Returns: false

hasValue({}, undefined);
// Returns: false

hasValue({ x: null }, null);
// Returns: true
```

## keyBy

Creates an object composed of keys generated from the results of running each element through a function.

```typescript
function keyBy<T>(arr: T[], iteratee: (item: T) => string): Record<string, T>
```

```typescript
import { keyBy } from 'es-toolkit';

const array = [
  { dir: 'left', code: 97 },
  { dir: 'right', code: 98 }
];

keyBy(array, (item) => item.dir);
// Returns: { 'left': { dir: 'left', code: 97 }, 'right': { dir: 'right', code: 98 } }

keyBy(['one', 'two', 'three'], (str) => str.length);
// Returns: { '3': 'two', '5': 'three' }
```

## map

Creates a new array with the results of calling a function on every element.

```typescript
function map<T, U>(arr: T[], iteratee: (item: T, index: number) => U): U[]
```

```typescript
import { map } from 'es-toolkit';

map([4, 8], (n) => n * n);
// Returns: [16, 64]

map([{ a: 1 }, { a: 2 }], 'a');
// Returns: [1, 2]

map(['a', 'b', 'c'], (char, index) => `${index}:${char}`);
// Returns: ['0:a', '1:b', '2:c']
```

## mapKeys

Creates an object with the same values as the original, but with keys generated by a function.

```typescript
function mapKeys<T>(obj: T, iteratee: (value: T[keyof T], key: keyof T) => string): Record<string, T[keyof T]>
```

```typescript
import { mapKeys } from 'es-toolkit';

mapKeys({ a: 1, b: 2 }, (value, key) => key + value);
// Returns: { 'a1': 1, 'b2': 2 }

mapKeys([1, 2, 3], (value) => `num_${value}`);
// Returns: { 'num_1': 1, 'num_2': 2, 'num_3': 3 }
```

## mapValues

Creates an object with the same keys as the original, but with values transformed by a function.

```typescript
function mapValues<T>(obj: T, iteratee: (value: T[keyof T]) => unknown): Record<keyof T, unknown>
```

```typescript
import { mapValues } from 'es-toolkit';

const users = {
  fred: { user: 'fred', age: 36 },
  pebbles: { user: 'pebbles', age: 1 }
};

mapValues(users, (o) => o.age);
// Returns: { fred: 36, pebbles: 1 }

mapValues({ a: 1, b: 2, c: 3 }, (value) => value * 2);
// Returns: { a: 2, b: 4, c: 6 }
```
